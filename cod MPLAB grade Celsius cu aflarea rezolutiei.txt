#define SIM_MPIDE

#include <stdio.h>
#include <stdlib.h>
#include <avr/io.h>
#include <avr/interrupt.h>
#include "Uno_Nano_utils.h"
#include "ATMega_Uart.h"
#include "ATMega_SPI.h"

#ifdef SIM_MPIDE
#define F_CPU 16000000UL
#define TIMER1_FREQUENCY_HZ 1000
#define STABLE_TIME_TICKS 3
#endif

volatile unsigned short TimebaseCount = 0;
volatile uint8_t Timebase_Tick = 0;

enum BtnStates { Released, Pressed };
volatile enum BtnStates BtnState = Released;
volatile unsigned short BTNTimeCount = 0;

FILE uart_str = FDEV_SETUP_STREAM(uart_putchar, uart_getchar, _FDEV_SETUP_WRITE);

void ConfigureandStartTimer1(uint16_t Frequency);

int main(int argc, char** argv) {
    uint8_t DataSend[2] = {0x00, 0x00};
    uint8_t DataRec[2] = {0x00, 0x00};
    uint16_t Temp_RAW = 0;
    int16_t Temp_Signed = 0;
    float Temp_Celsius = 0.0;
    float Resolution_Step = 0.0625; // default (12-bit)

    stdout = stdin = &uart_str;

    ConfigureUart();
    SPISetup();
    ConfigureandStartTimer1(TIMER1_FREQUENCY_HZ);

    BitSet(PCMSK2, PCINT21);  // Buton PD5
    BitSet(PCICR, PCIE2);     // Activare întreruperi pin change

    sei();

    printf("\r\nConfiguring DS1626, press BTN to continue...");
    while (BtnState != Pressed);
    while (BtnState == Pressed);
    printf("\r\nBTN Pressed");

    //START CONVERT
    BitSet(PORT_SS, SS_PIN);
    DataSend[0] = 0x51;
    SPITransfer(DataSend, DataRec, 1);
    BitClear(PORT_SS, SS_PIN);

    //READ CONFIGURATION REGISTER (0xAC)
    printf("\r\nReading configuration register...");
    BitSet(PORT_SS, SS_PIN);
    DataSend[0] = 0xAC;
    SPITransfer(DataSend, DataRec, 1);
    BitClear(DDRB, PB3);  // MOSI -> input
    DataSend[0] = 0x00;
    SPITransfer(DataSend, DataRec, 1);
    BitSet(DDRB, PB3);    // MOSI -> output
    BitClear(PORT_SS, SS_PIN);

    uint8_t configByte = DataRec[0];
    uint8_t R1 = (configByte >> 2) & 0x01;
    uint8_t R0 = (configByte >> 1) & 0x01;

    if (R1 == 0 && R0 == 0) Resolution_Step = 0.5;
    else if (R1 == 0 && R0 == 1) Resolution_Step = 0.25;
    else if (R1 == 1 && R0 == 0) Resolution_Step = 0.125;
    else Resolution_Step = 0.0625;

    printf("\r\nConfig byte: 0x%02X | R1=%d, R0=%d => Resolution: %.4f °C/bit",
            configByte, R1, R0, Resolution_Step);

    while (1) {
        if (BtnState == Pressed || Timebase_Tick) {
            printf("\r\nReading temperature...");

            //READ TEMPERATURE
            BitSet(PORT_SS, SS_PIN);
            DataSend[0] = 0xAA;
            SPITransfer(DataSend, DataRec, 1);

            BitClear(DDRB, PB3); // MOSI -> input
            DataSend[0] = 0x00;
            DataSend[1] = 0x00;
            SPITransfer(DataSend, DataRec, 2);
            BitSet(DDRB, PB3);   // MOSI -> output
            BitClear(PORT_SS, SS_PIN);

            Temp_RAW = (DataRec[1] << 8) | DataRec[0];
            Temp_Signed = (int16_t)Temp_RAW;
            Temp_Celsius = Temp_Signed * Resolution_Step;

            printf("\r\nTEMP_RAW = 0x%04X", Temp_RAW);
            printf("\r\nTemperature = %.4f °C", Temp_Celsius);

            if (Timebase_Tick) Timebase_Tick = 0;
            else while (BtnState == Pressed);
        }
    }

    return 0;
}

ISR(TIMER1_COMPA_vect) {
    BTNTimeCount++;
    TimebaseCount++;
    if (TimebaseCount == 1000) {
        TimebaseCount = 0;
        Timebase_Tick = 1;
    }
}

ISR(PCINT2_vect) {
    if (BTNTimeCount >= STABLE_TIME_TICKS) {
        if (bit_is_set(PIND, PIND5)) BtnState = Pressed;
        else BtnState = Released;
    }
    BTNTimeCount = 0;
}

void ConfigureandStartTimer1(uint16_t Frequency) {
    TCCR1A = 0;
    TCCR1B = 0;
    TCNT1 = 0;
    OCR1A = 16000 - 1;
    BitSet(TCCR1B, WGM12);    // CTC mode
    BitSet(TIMSK1, OCIE1A);   // Compare interrupt
    BitSet(TCCR1B, CS10);     // Prescaler = 1
}
